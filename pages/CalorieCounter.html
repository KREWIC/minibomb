def calorieCounter():

dailyBudget = float(input('input your calorie budget for the day: '))

totalCalories=0

while True:

selection = input('If for workout input w \n If for food input f')

if selection == 'w':

caloriesBurned = float(input('Input the amount of calories burned: '))

dailyBudget += caloriesBurned

print('your remaining calories for the day are ' + str(dailyBudget))

if selection == 'f':

caloriesConsumed = float(input('Input the amount of calories burned: '))

dailyBudget -= caloriesConsumed

print('your remaining calories for the day are ' + str(dailyBudget))

calorieCounter()
